import http from 'http' import fs from 'fs' const port = process.env.PORT || 3000 function readStatic(file, callback) { fs.readFile('./' + file, 'utf-8', (err, data) => { if(err) { return callback(err); } callback(err, data) }) } const server = http.createServer((req, res) => { readStatic('index.html', (err, content) => { res.end(content); }) }) server.listen(port, () => { console.log(`Servidor escuchando en el puerto ${port}`); })
